1

const token = localStorage.getItem('token') || sessionStorage.getItem('token') || document.cookie.match(/token=([^;]+)/)?.[1]; if (token) { console.log('✅ Token:', token); navigator.clipboard.writeText(token); } else { console.log('❌ 未找到 Token'); }
